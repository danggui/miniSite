alert-控件

基本参数说明

@param title        {String}    标题
@param content      {String}    内容
@param buttons      {array}     需要生成的按钮
@param template     {String}    模板视图文本
@param events       {Object}    绑定事件
@param setDatamodel {Object}    动态设置属性、事件

demo1

if (!this.alert1) {
this.alert1 = new UIAlert();
}
this.alert1.animateShow();

demo2

if (!this.alert2) {
this.alert2 = new UIAlert({
datamodel: {
title: 'demo01',
content: 'content01'
},
okAction: function () {
alert('ok');
this.hide();
},
cancelAction: function () {
alert('cancel');
this.hide();
}
});
}
this.alert2.show();

demo3

if (!this.alert3) {
this.alert3 = new UIAlert({
datamodel: {
title: 'demo02',
content: 'content02',
btns: [
{ name: '2', className: 'ct-ui-btns-ok' },
{ name: '22', className: 'ct-ui-btns-no' },
{ name: '222', className: 'ct-ui-btns-unknown' }
]
},
events: {
'click .ct-ui-btns-ok': 'okAction',
'click .ct-ui-btns-no': 'noAction',
'click .ct-ui-btns-unknown': 'unknownAction'
},
okAction: function () {
alert('2');
this.hide();
},
noAction: function () {
alert('22');
this.hide();
},
unknownAction: function () {
alert('222');
this.hide();
}
});
}
this.alert3.show();
