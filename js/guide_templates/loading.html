loading-控件

基本参数说明

@param template     {String}    自定义模板
@param maskToHide   {Bool}      是否点击mask隐藏
@param needMask     {Bool}      是否需要mask

demo1

if (!this.loading1) {
this.loading1 = new UILoading({
maskToHide: true
});
}
this.loading1.show();

demo3

if (!this.loading3) {
this.loading3 = new UILoading({
needMask: false
});
}
this.loading3.show();
